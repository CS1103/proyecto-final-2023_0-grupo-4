65void Maze::GenerateMaze(uint8_t height, uint8_t width, HUNT_METHOD huntMethod) {
66
67 // generate matrix of walls
68 m_maze = std::vector<std::vector<SQUARE_TYPE>>(
69 height, std::vector<SQUARE_TYPE>(width, SQUARE_TYPE::WALL));
70
71 uint8_t start_x = 0;
72 uint8_t start_y = 0;
73
74 do {
75 start_x = Utils::RandomNum<uint8_t>(1, height - 1);
76 } while (start_x % 2 != 1);
77 do {
78 start_y = Utils::RandomNum<uint8_t>(1, width - 1);
79 } while (start_y % 2 != 1);
80
81 m_maze[start_x][start_y] = SQUARE_TYPE::EMPTY;
82
83 uint8_t visit_count = 1;
84
85 // constexpr HUNT_METHOD METHOD = HUNT_METHOD::RANDOM;
86
87 if (huntMethod == HUNT_METHOD::RANDOM) {
88
89 std::optional<square> first_hunt_opt =
90 Hunt<HUNT_METHOD::RANDOM>(visit_count);
91 while (first_hunt_opt.has_value()) {
92 auto [row, col] = first_hunt_opt.value();
93 auto walk = RandomWalk({row, col});
94
95 visit_count += SolveRandomWalk(walk, {row, col});
96 first_hunt_opt = Hunt<HUNT_METHOD::RANDOM>(visit_count);
97 }
98
99 } else if (huntMethod == HUNT_METHOD::SERPENTINE) {
100
101 std::optional<square> first_hunt_opt =
102 Hunt<HUNT_METHOD::SERPENTINE>(std::nullopt);
103 while (first_hunt_opt.has_value()) {
104 auto [row, col] = first_hunt_opt.value();
105 auto walk = RandomWalk({row, col});
106
107 visit_count += SolveRandomWalk(walk, {row, col});
108 first_hunt_opt = Hunt<HUNT_METHOD::SERPENTINE>(std::nullopt);
109 }
110 }
111 do {
112 start_x = Utils::RandomNum<uint8_t>(1, height - 1);
113 start_y = Utils::RandomNum<uint8_t>(1, width - 1);
114 } while (m_maze[start_x][start_y] != SQUARE_TYPE::EMPTY);
115 m_maze[start_x][start_y] = SQUARE_TYPE::START;
116 m_start = {start_x, start_y};
117
118 uint8_t end_x = 0;
119 uint8_t end_y = 0;
120
121 do {
122 end_x = Utils::RandomNum<uint8_t>(1, height - 1);
123 end_y = Utils::RandomNum<uint8_t>(1, width - 1);
124 } while (m_maze[end_x][end_y] != SQUARE_TYPE::EMPTY);
125 m_maze[end_x][end_y] = SQUARE_TYPE::GOAL;
126 m_goal = {end_x, end_y};
127}

175std::optional<square>
176Maze::Hunt<HUNT_METHOD::SERPENTINE>(const std::optional<uint8_t> &count) {
177
178 if (count.has_value()) {
179 throw std::runtime_error("Count must be empty with SERPENTINE hunt method");
180 }
181
182 std::pair<int8_t, int8_t> cell = {1, -1};
183 bool found = false;
184
185 while (!found) {
186 cell.second += 2;
187 if (cell.second > static_cast<int8_t>(m_maze[0].size() - 2)) {
188 cell.first += 2;
189 cell.second = 1;
190 if (cell.first > static_cast<int8_t>(m_maze.size() - 2)) {
191 return std::nullopt;
192 }
193 }
194
195 if (m_maze[static_cast<uint8_t>(cell.first)]
196 [static_cast<uint8_t>(cell.second)] != SQUARE_TYPE::EMPTY) {
197 found = true;
198 }
199 }
200
201 // if is valid cell
202 if (cell.first >= 0 && cell.second >= 0 &&
203 cell.first < static_cast<int8_t>(m_maze.size()) &&
204 cell.second < static_cast<int8_t>(m_maze[0].size())) {
205 return std::make_pair(static_cast<uint8_t>(cell.first),
206 static_cast<uint8_t>(cell.second));
207 }
208 throw std::runtime_error("Invalid cell");
209}

212size_t Maze::HashPair::operator()(const std::pair<T1, T2> &pair) const {
213 // https://www.geeksforgeeks.org/how-to-create-an-unordered_map-of-pairs-in-c/
214 auto hash1 = std::hash<T1>{}(pair.first);
215 auto hash2 = std::hash<T2>{}(pair.second);
216
217 if (hash1 != hash2) {
218 return hash1 ^ hash2;
219 }
220
221 return hash1;

267uint8_t Maze::SolveRandomWalk(
268 const std::unordered_map<square, direction_t, HashPair> &walk,
269 const square &start) {
270
271 uint8_t count = 0;
272
273 square current = start;
274
275 while (m_maze[current.first][current.second] != SQUARE_TYPE::EMPTY) {
276 m_maze[current.first][current.second] = SQUARE_TYPE::EMPTY;
277 square next = Move(current, walk.at(current));
278 uint8_t new_x = (next.first + current.first) / 2;
279 uint8_t new_y = (next.second + current.second) / 2;
280 m_maze[new_x][new_y] = SQUARE_TYPE::EMPTY;
281 count++;
282 current = next;
283 }
284 return count;
285}
